078/**
079 * {@link Artifact} to {@link LicenseSet}
080 * ({@link org.spdx.rdfparser.license.License}) {@link java.util.Map}
081 * implementation.  The {@link #get(Object)} method transparently calculates
082 * and caches any value.
083 *
084 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
085 */
086@Named @Singleton
087@Slf4j
088public class ArtifactLicenseCatalog extends TreeMap<Artifact,AnyLicenseInfo> {
089    private static final long serialVersionUID = -7887839577334232433L;
090
091    private static final String CATALOG = "artifact-license-catalog.xml";
092
093    private static final int FLUSH_PERIOD = 8;
094
095    private static final Predicate<String> INCLUDE =
096        Pattern.compile("(?i)^(.*/|)(LICENSE([.][^/]+)?|about.html)$")
097        .asPredicate();
098    private static final Predicate<String> EXCLUDE =
099        Pattern.compile("(?i)^.*[.]class$")
100        .asPredicate().negate();
101
102    private static final Comparator<? super Boolean> TRUTH =
103        (t, u) -> Objects.equals(t, u) ? 0 : (t ? -1 : 1);
104    private static Comparator<? super AnyLicenseInfo> SIEVE =
105        Comparator
106        .<AnyLicenseInfo,Boolean>
107        comparing(t -> (! (t instanceof URLLicenseInfo)), TRUTH)
108        .thenComparing(t -> (! (t instanceof TextLicenseInfo)), TRUTH)
109        .thenComparing(t -> (! (t instanceof ExtractedLicenseInfo)), TRUTH)
110        .thenComparing(LicenseUtilityMethods::isFullySpdxListed, TRUTH)
111        .thenComparing(t -> (t instanceof LicenseSet), TRUTH)
112        .thenComparing(LicenseUtilityMethods::countOf, Comparator.<Integer>reverseOrder())
113        .thenComparing(t -> (t instanceof WithExceptionOperator), TRUTH)
114        .thenComparing(t -> (t instanceof OrLaterOperator), TRUTH)
115        .thenComparing(LicenseUtilityMethods::isPartiallySpdxListed, TRUTH);
116
117    /** @serial */ private final MavenSession session;
118    /** @serial */ private final ArtifactModelCache cache;
119    /** @serial */ private final LicenseMap map;
120    /** @serial */ private final LicenseResolver resolver;
121    /** @serial */ private final File file;
122    /** @serial */ private final Properties defaults = new Properties();
123    /** @serial */ private final Properties catalog = new Properties(defaults);

125    /**
126     * Sole constructor.
127     *
128     * @param   session         The injected {@link MavenSession}.
129     * @param   cache           The injected {@link ArtifactModelCache}.
130     * @param   map             The injected {@link LicenseMap}.
131     * @param   resolver        The injected {@link LicenseResolver}.
132     */
133    @Inject
134    public ArtifactLicenseCatalog(MavenSession session, ArtifactModelCache cache, LicenseMap map, LicenseResolver resolver) {
135        super(Comparator.comparing(ArtifactUtils::key, String.CASE_INSENSITIVE_ORDER));
136
137        this.session = Objects.requireNonNull(session);
138        this.cache = Objects.requireNonNull(cache);
139        this.map = Objects.requireNonNull(map);
140        this.resolver = Objects.requireNonNull(resolver);
141        this.file = new File(session.getLocalRepository().getBasedir(), CATALOG);
142    }
143
144    protected void load() {
145        try (InputStream in = getClass().getResourceAsStream(CATALOG)) {
146            if (in != null) {
147                defaults.loadFromXML(in);
148            }
149        } catch (IOException exception) {
150        }
151
152        if (file.exists()) {
153            try (FileInputStream in = new FileInputStream(file)) {
154                catalog.loadFromXML(in);
155            } catch (IOException exception) {
156                log.error("Cannot read {}", file);
157            }
158        }
159
160        for (String key : catalog.stringPropertyNames()) {
161            try {
162                put(new KeyArtifact(key), resolver.parseLicenseString(catalog.getProperty(key)));
163            } catch (Exception exception) {
164                log.error("{}: {}", key, exception.getMessage(), exception);
165            }
166        }
167    }
168
169    protected void flush() {
170        boolean dirty = (! file.exists());
171
172        for (Map.Entry<Artifact,AnyLicenseInfo> entry : entrySet()) {
173            AnyLicenseInfo license = entry.getValue();
174
175            if (isFullySpdxListed(license)) {
176                String key = ArtifactUtils.key(entry.getKey());
177                String value = license.toString();
178
179                dirty |= (! Objects.equals(value, catalog.put(key, value)));
180            }
181        }
182
183        if (dirty) {
184            try (FileOutputStream out = new FileOutputStream(file)) {
185                catalog.storeToXML(out, file.getName());
186            } catch (IOException exception) {
187                log.warn("Cannot write {}", file);
188            }
189        }
190    }
191
192    @PostConstruct
193    public void init() {
194        load();
195        Runtime.getRuntime().addShutdownHook(new Thread(() -> flush()));
196    }
197
198    @PreDestroy
199    public void destroy() {
200        flush();
201        log.debug("{}.size() = {}", getClass().getSimpleName(), size());
202    }
203
204    @Override
205    public AnyLicenseInfo get(Object key) {
206        AnyLicenseInfo value = super.get(key);
207
208        if (value == null) {
209            value = compute((Artifact) key);
210
211            put((Artifact) key, value);
212
213            if ((size() % FLUSH_PERIOD) == 0) {
214                flush();
215            }
216        }
217
218        return value;
219    }

221    private AnyLicenseInfo compute(Artifact artifact) {
222        URL url = toURL(artifact);
223        /*
224         * Licenses specified in the Manifest Bundle-License
225         * or found in the Artifact
226         */
227        List<AnyLicenseInfo> bundle = Collections.emptyList();
228        List<AnyLicenseInfo> scanned = Collections.emptyList();
229
230        try {
231            JarFile jar = ((JarURLConnection) url.openConnection()).getJarFile();
232            Pattern pattern = Pattern.compile("((?<id>.+);link=)?(?<url>.*)");
233
234            bundle =
235                Stream.of(jar.getManifest())
236                .filter(Objects::nonNull)
237                .map(t -> t.getMainAttributes().getValue("Bundle-License"))
238                .filter(StringUtils::isNotBlank)
239                .flatMap(t -> Stream.of(t.split(",")))
240                .map(t -> t.trim())
241                .distinct()
242                .map(pattern::matcher)
243                .filter(Matcher::matches)
244                .map(t -> parse(t.group("id"), resolve(url, t.group("url"))))
245                .collect(toList());
246
247            scanned =
248                jar.stream()
249                .map(JarEntry::getName)
250                .filter(INCLUDE)
251                .filter(EXCLUDE)
252                .map(t -> parse(t, resolve(url, t)))
253                .filter(t -> (! (t instanceof URLLicenseInfo)))
254                .collect(toList());
255        } catch (ZipException exception) {
256        } catch (IOException exception) {
257            log.debug("{}: {}", artifact, exception.getMessage(), exception);
258        }
259
260        Map<String,AnyLicenseInfo> found =
261            scanned.stream()
262            .filter(t -> LicenseUtilityMethods.countOf(t) > 0)
263            .filter(LicenseUtilityMethods::isFullySpdxListed)
264            .collect(toMap(k -> k.toString(), v -> v, (t, u) -> t));
265        /*
266         * Licenses specified in the Artifact's POM
267         */
268        List<AnyLicenseInfo> pom = Collections.emptyList();
269        Model model = cache.get(artifact);
270
271        if (model != null) {
272            pom =
273                Stream.of(model.getLicenses())
274                .filter(Objects::nonNull)
275                .flatMap(List::stream)
276                .distinct()
277                .map(t -> parse((t.getName() != null) ? t.getName().replaceAll(",", "") : null,
278                                resolve(toURL(model.getUrl()), t.getUrl())))
279                .collect(toList());
280        }
281
282        if (bundle.isEmpty() && pom.isEmpty() && found.isEmpty()) {
283            log.warn("{}: No license(s) specified or found", artifact);
284        }
285
286        List<AnyLicenseInfo> licenses = bundle.stream().collect(toList());
287
288        if (! isFullySpecified(licenses)) {
289            if (licenses.isEmpty() || isFullySpecified(pom)) {
290                licenses.clear();
291                licenses.addAll(pom);
292            }
293        }
294
295        if ((! isFullySpecified(licenses)) && found.size() >= licenses.size()) {
296            licenses.clear();
297            licenses.addAll(found.values());
298        }
299
300        if ((! licenses.isEmpty()) && (! isFullySpecified(licenses))) {
301            log.debug("------------------------------------------------------------");
302            log.debug("{}", ArtifactUtils.key(artifact));
303            log.debug("{}", url);
304            log.debug("      Bundle: {}", bundle);
305            log.debug("         POM: {}", pom);
306            log.debug("     Scanned: {}", scanned);
307            log.debug("       Found: {}", found.values());
308            log.debug("  License(s): {}", licenses);
309            log.debug("------------------------------------------------------------");
310        }
311
312        return resolver.toLicense(licenses);
313    }

315    private URL toURL(Artifact artifact) {
316        URL url = null;
317
318        try {
319            url = new URI("jar", artifact.getFile().toURI().toASCIIString() + "!/", null).toURL();
320        } catch(URISyntaxException | MalformedURLException exception) {
321            log.debug("{}: {}", artifact, exception.getMessage(), exception);
322            throw new IllegalStateException(exception);
323        }
324
325        return url;
326    }
327
328    private URL toURL(String string) {
329        URL url = null;
330
331        if (isNotBlank(string)) {
332            try {
333                url = new URI(string).toURL();
334            } catch(URISyntaxException | MalformedURLException exception) {
335                log.debug("{}: {}", string, exception.getMessage(), exception);
336            }
337        }
338
339        return url;
340    }

342    private String[] resolve(URL root, String urls) {
343        ArrayList<String> list = new ArrayList<>();
344
345        if (isNotBlank(urls)) {
346            for (String url : urls.split("[,\\p{Space}]+")) {
347                if (isNotBlank(url)) {
348                    String value = url;
349
350                    if (root != null && (! isAbsolute(value))) {
351                        String string = root.toString();
352
353                        if (! string.endsWith("/")) {
354                            string += "/";
355                        }
356
357                        value = string + value;
358                    }
359
360                    list.add(value);
361                }
362            }
363        }
364
365        return list.toArray(new String[] { });
366    }
367
368    private boolean isAbsolute(String url) {
369        boolean isAbsolute = false;
370
371        try {
372            isAbsolute = new URI(url).isAbsolute();
373        } catch (Exception exception) {
374        }
375
376        return isAbsolute;
377    }

379    private AnyLicenseInfo parse(String id, String... urls) {
380        AnyLicenseInfo option0 = isNotBlank(id) ? map.get(id) : null;
381        AnyLicenseInfo option1 = null;
382
383        try {
384            option1 = isNotBlank(id) ? resolver.parseLicenseString(id) : null;
385        } catch (Exception exception) {
386        }
387
388        AnyLicenseInfo option2 = new URLLicenseInfo(isNotBlank(id) ? id : urls[0], urls);
389        AnyLicenseInfo[] options =
390            Stream.of(option0, option1, option2)
391            .filter(Objects::nonNull)
392            .map(t -> resolver.parse(t))
393            .toArray(AnyLicenseInfo[]::new);
394
395        Arrays.sort(options, SIEVE);
396
397        return (options.length > 0) ? options[0] : null;
398    }
399
400    private int countOf(Collection<? extends AnyLicenseInfo> collection) {
401        return collection.stream().mapToInt(LicenseUtilityMethods::countOf).sum();
402    }
403
404    private boolean isFullySpecified(Collection<AnyLicenseInfo> collection) {
405        return ((! collection.isEmpty()) && collection.stream().allMatch(t -> isFullySpdxListed(t)));
406    }

408    private class KeyArtifact extends DefaultArtifact {
409        public KeyArtifact(String gav) { this(gav.split(":", 3)); }
410
411        private KeyArtifact(String[] gav) {
412            super(gav[0], gav[1], gav[2], EMPTY, "pom", EMPTY, null);
413        }
414    }

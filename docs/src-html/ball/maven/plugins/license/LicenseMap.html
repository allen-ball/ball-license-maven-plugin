021import com.fasterxml.jackson.databind.JsonNode;
022import com.fasterxml.jackson.databind.ObjectMapper;
023import java.util.Objects;
024import java.util.Properties;
025import java.util.TreeMap;
026import java.util.stream.Stream;
027import javax.annotation.PostConstruct;
028import javax.annotation.PreDestroy;
029import javax.inject.Inject;
030import javax.inject.Named;
031import javax.inject.Singleton;
032import lombok.extern.slf4j.Slf4j;
033import org.apache.commons.lang3.StringUtils;
034import org.spdx.rdfparser.license.AnyLicenseInfo;
035import org.spdx.rdfparser.license.License;
036import org.spdx.rdfparser.license.LicenseInfoFactory;
037/* import org.spdx.rdfparser.license.ListedExceptions; */
038import org.spdx.rdfparser.license.ListedLicenses;
039
040import static ball.maven.plugins.license.LicenseUtilityMethods.isFullySpdxListed;
041import static org.apache.commons.lang3.StringUtils.SPACE;
042
043/**
044 * {@link java.util.Map} implementation that relates SPDX ID and known
045 * aliases to {@link License} sourced from
046 * {@link ListedLicenses#getListedLicenses()}.
047 *
048 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
049 */
050@Named @Singleton
051@Slf4j
052public class LicenseMap extends TreeMap<String,AnyLicenseInfo> implements DefaultMethods {
053    private static final long serialVersionUID = 8959458091372664080L;
054
055    private static final String ONLY_USE_LOCAL_LICENSES = "SPDXParser.OnlyUseLocalLicenses";
056
057    /**
058     * {@code resources/licenses-full.json} read from SPDX artifacts.
059     */
060    public static final JsonNode LICENSES_FULL_JSON;
061
062    static {
063        if (System.getProperty(ONLY_USE_LOCAL_LICENSES) == null) {
064            System.setProperty(ONLY_USE_LOCAL_LICENSES, String.valueOf(true));
065        }
066
067        try {
068            LICENSES_FULL_JSON =
069                new ObjectMapper()
070                .readTree(LicenseMap.class.getClassLoader().getResource("resources/licenses-full.json"));
071        } catch (Exception exception) {
072            throw new ExceptionInInitializerError(exception);
073        }
074    }
075
076    /**
077     * Sole constructor.
078     */
079    @Inject
080    public LicenseMap() { super(String.CASE_INSENSITIVE_ORDER); }
081
082    @PostConstruct
083    public void init() throws Exception {
084        ListedLicenses licenses = ListedLicenses.getListedLicenses();
085        /* ListedExceptions exceptions = ListedExceptions.getListedExceptions(); */
086
087        for (String key : licenses.getSpdxListedLicenseIds()) {
088            License license = licenses.getListedLicenseById(key);
089
090            put(key, license);
091            putIfAbsent(license.getName(), license);
092        }
093
094        for (JsonNode node : LICENSES_FULL_JSON.at("/licenses")) {
095            String spdx = node.at("/identifiers/spdx/0").asText();
096
097            if (containsKey(spdx)) {
098                Stream.of(node.at("/id").asText(), node.at("/name").asText())
099                    .map(t -> t.trim())
100                    .map(t -> t.replaceAll("[\\p{Space}]+", SPACE))
101                    .filter(StringUtils::isNotBlank)
102                    .forEach(t -> computeIfAbsent(t, k -> get(spdx)));
103            }
104        }
105
106        Properties aliases = getXMLProperties("aliases");
107
108        for (String id : aliases.stringPropertyNames()) {
109            AnyLicenseInfo value = LicenseInfoFactory.parseSPDXLicenseString(id);
110
111            if (! isFullySpdxListed(value)) {
112                throw new IllegalArgumentException(id);
113            }
114
115            Stream.of(aliases.getProperty(id).split("\\R+"))
116                .map(String::trim)
117                .filter(StringUtils::isNotBlank)
118                .forEach(t -> putIfAbsent(t, value));
119        }
120    }
121
122    @PreDestroy
123    public void destroy() {
124        log.debug("{}.size() = {}", getClass().getSimpleName(), size());
125    }
126
127    @Override
128    public AnyLicenseInfo get(Object key) {
129        AnyLicenseInfo value = super.get(key);
130
131        if (value == null) {
132            if (key instanceof String) {
133                String string = ((String) key).trim();
134
135                value =
136                    Stream.of(string.replaceAll("(?i)[\\p{Space}]+", "-"),
137                              string.replaceAll("(?i)(V(ERSION)?)[\\p{Space}]?([\\p{Digit}])", "$3"))
138                    .filter(t -> (! string.equals(t)))
139                    .map(t -> get(t))
140                    .filter(Objects::nonNull)
141                    .findFirst().orElse(null);
142            }
143        }
144
145        return value;
146    }
147}

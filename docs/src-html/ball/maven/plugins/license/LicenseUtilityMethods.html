021import java.util.Arrays;
022import java.util.Collection;
023import java.util.Collections;
024import java.util.Iterator;
025import java.util.Objects;
026import java.util.Spliterator;
027import java.util.Spliterators.AbstractSpliterator;
028import java.util.function.Consumer;
029import java.util.function.Function;
030import java.util.function.Supplier;
031import java.util.stream.Stream;
032import java.util.stream.StreamSupport;
033import lombok.NoArgsConstructor;
034import lombok.ToString;
035import org.spdx.rdfparser.license.AnyLicenseInfo;
036import org.spdx.rdfparser.license.License;
037import org.spdx.rdfparser.license.LicenseSet;
038import org.spdx.rdfparser.license.OrLaterOperator;
039import org.spdx.rdfparser.license.WithExceptionOperator;
040
041import static lombok.AccessLevel.PRIVATE;
042
043/**
044 * {@link AnyLicenseInfo} utility methods.
045 *
046 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
047 */
048@NoArgsConstructor(access = PRIVATE) @ToString
049public abstract class LicenseUtilityMethods {
050
051    /**
052     * Static method to create a {@link Stream} to walk the tree rooted at
053     * {@code root}.
054     *
055     * @param   root            The {@link AnyLicenseInfo} representing the
056     *                          root of the tree.
057     *
058     * @return  A {@link Stream}.
059     */
060    public static Stream<AnyLicenseInfo> walk(AnyLicenseInfo root) {
061        return walk(root, LicenseUtilityMethods::childrenOf);
062    }
063
064    /**
065     * Static method to get the children of a {@link AnyLicenseInfo}.
066     *
067     * @param   node            The {@link AnyLicenseInfo}.
068     *
069     * @return  A {@link Collection} of {@link AnyLicenseInfo} for a
070     *          {@link LicenseSet}, {@link OrLaterOperator}, or
071     *          {@link WithExceptionOperator} node; an empty
072     *          {@link Collection} otherwise.
073     */
074    public static Collection<AnyLicenseInfo> childrenOf(AnyLicenseInfo node) {
075        AnyLicenseInfo[] children = null;
076
077        if (node instanceof LicenseSet) {
078            children = ((LicenseSet) node).getMembers();
079        } else if (node instanceof OrLaterOperator) {
080            children = new AnyLicenseInfo[] { ((OrLaterOperator) node).getLicense() };
081        } else if (node instanceof WithExceptionOperator) {
082            children = new AnyLicenseInfo[] { ((WithExceptionOperator) node).getLicense() };
083        } else {
084            children = new AnyLicenseInfo[] { };
085        }
086
087        return Arrays.asList(children);
088    }
089
090    /**
091     * Returns the count of licenses represented by this
092     * {@link AnyLicenseInfo}: The number of members in a
093     * {@link LicenseSet} or {@code 1}.
094     *
095     * @param   license         The {@link AnyLicenseInfo}.
096     *
097     * @return  The license count.
098     */
099    public static int countOf(AnyLicenseInfo license) {
100        int count = 0;
101
102        if (license instanceof LicenseSet) {
103            count =
104                Stream.of(((LicenseSet) license).getMembers())
105                .mapToInt(t -> countOf(t))
106                .sum();
107        } else if (license != null) {
108            count = 1;
109        }
110
111        return count;
112    }
113
114    /**
115     * Method to determine if an {@link AnyLicenseInfo} is "empty" (defined
116     * as the license consists solely of empty containers).
117     *
118     * @param   license         The {@link AnyLicenseInfo}.
119     *
120     * @return  {@code true} if the {@code license} is empty; {@code false}
121     *          otherwise.
122     */
123    public static boolean isEmpty(AnyLicenseInfo license) {
124        return countOf(license) == 0;
125    }
126
127    /**
128     * Static method to test if an {@link AnyLicenseInfo} is a branch.
129     *
130     * @param   node            The {@link AnyLicenseInfo}.
131     *
132     * @return  {@code true} if a branch; {@code false} otherwise.
133     */
134    public static boolean isBranch(AnyLicenseInfo node) {
135        return (isLicenseSet(node) || isOperator(node));
136    }
137
138    /**
139     * Static method to test if an {@link AnyLicenseInfo} is a leaf.
140     *
141     * @param   node            The {@link AnyLicenseInfo}.
142     *
143     * @return  {@code true} if a leaf; {@code false} otherwise.
144     */
145    public static boolean isLeaf(AnyLicenseInfo node) {
146        return (! isBranch(node));
147    }
148
149    /**
150     * Static method to test if an {@link AnyLicenseInfo}
151     * is a {@link LicenseSet}.
152     *
153     * @param   node            The {@link AnyLicenseInfo}.
154     *
155     * @return  {@code true} if a {@code node} is a {@link LicenseSet};
156     *          {@code false} otherwise.
157     */
158    public static boolean isLicenseSet(AnyLicenseInfo node) {
159        return (node instanceof LicenseSet);
160    }
161
162    /**
163     * Static method to test if an {@link AnyLicenseInfo} is an operator
164     * node ({@link OrLaterOperator} or {@link WithExceptionOperator}).
165     *
166     * @param   node            The {@link AnyLicenseInfo}.
167     *
168     * @return  {@code true} if an operator node; {@code false} otherwise.
169     */
170    public static boolean isOperator(AnyLicenseInfo node) {
171        return (node instanceof OrLaterOperator || node instanceof WithExceptionOperator);
172    }
173
174    /**
175     * Method to test if an {@link AnyLicenseInfo} is fully specified by
176     * SPDX license(s).
177     *
178     * @param   license         The {@link AnyLicenseInfo}.
179     *
180     * @return  {@code true} if fully specified; {@code false} otherwise.
181     */
182    public static boolean isFullySpdxListed(AnyLicenseInfo license) {
183        boolean fullySpecified =
184             LicenseUtilityMethods.walk(license)
185            .filter(LicenseUtilityMethods::isLeaf)
186            .map(t -> (t instanceof License))
187            .reduce(Boolean::logicalAnd).orElse(false);
188
189        return fullySpecified;
190    }
191
192    /**
193     * Method to test if an {@link AnyLicenseInfo} is partially specified by
194     * SPDX license(s).
195     *
196     * @param   license         The {@link AnyLicenseInfo}.
197     *
198     * @return  {@code true} if partially specified; {@code false}
199     *          otherwise.
200     */
201    public static boolean isPartiallySpdxListed(AnyLicenseInfo license) {
202        boolean partiallySpecified =
203            LicenseUtilityMethods.walk(license)
204            .filter(LicenseUtilityMethods::isLeaf)
205            .map(t -> (t instanceof License))
206            .reduce(Boolean::logicalOr).orElse(false);
207
208        return partiallySpecified;
209    }
210
211    private static <T> Stream<T> walk(T root, Function<? super T,Collection<? extends T>> childrenOf) {
212        Walker<T> walker = new Walker<>(Collections.singleton(root), childrenOf);
213
214        return StreamSupport.stream(walker, false);
215    }
216
217    @ToString
218    private static class Walker<T> extends AbstractSpliterator<T> {
219        private final Stream<Supplier<Walker<T>>> stream;
220        private Iterator<Supplier<Walker<T>>> iterator = null;
221        private Spliterator<? extends T> spliterator = null;
222
223        private Walker(Collection<? extends T> nodes, Function<? super T,Collection<? extends T>> childrenOf) {
224            super(Long.MAX_VALUE, IMMUTABLE | NONNULL);
225
226            stream =
227                nodes.stream()
228                .filter(Objects::nonNull)
229                .map(childrenOf)
230                .filter(Objects::nonNull)
231                .map(t -> (() -> new Walker<T>(t, childrenOf)));
232            spliterator =
233                nodes.stream()
234                .filter(Objects::nonNull)
235                .spliterator();
236        }
237
238        @Override
239        public boolean tryAdvance(Consumer<? super T> consumer) {
240            boolean accepted = false;
241
242            if (spliterator != null) {
243                accepted = spliterator.tryAdvance(consumer);
244            }
245
246            if (! accepted) {
247                if (iterator == null) {
248                    iterator = stream.iterator();
249                }
250
251                if (iterator.hasNext()) {
252                    spliterator = iterator.next().get();
253                    accepted = tryAdvance(consumer);
254                }
255            }
256
257            return accepted;
258        }
259    }

021import java.io.File;
022import java.io.FileInputStream;
023/* import java.io.FileOutputStream; */
024import java.util.List;
025import java.util.Objects;
026import java.util.stream.Stream;
027import javax.inject.Inject;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.slf4j.Slf4j;
031import org.apache.maven.model.Model;
032import org.apache.maven.model.io.xpp3.MavenXpp3Reader;
033/* import org.apache.maven.model.io.xpp3.MavenXpp3Writer; */
034import org.apache.maven.plugin.MojoExecutionException;
035import org.apache.maven.plugin.MojoFailureException;
036import org.apache.maven.plugins.annotations.Mojo;
037import org.apache.maven.plugins.annotations.Parameter;
038import org.apache.maven.project.MavenProject;
039import org.jsoup.Jsoup;
040import org.jsoup.nodes.Document;
041import org.spdx.rdfparser.license.AnyLicenseInfo;
042import org.spdx.rdfparser.license.License;
043
044import static java.util.stream.Collectors.toList;
045import static org.apache.commons.lang3.StringUtils.isBlank;
046import static org.apache.maven.plugins.annotations.LifecyclePhase.INITIALIZE;
047import static org.spdx.compare.LicenseCompareHelper.isTextStandardLicense;
048
049/**
050 * {@link org.apache.maven.plugin.Mojo} to update project LICENSE.
051 *
052 * {@injected.fields}
053 *
054 * {@maven.plugin.fields}
055 *
056 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
057 */
058@Mojo(name = "update-project-license", configurator = "license-mojo-component-configurator",
059      defaultPhase = INITIALIZE, requiresProject = true)
060@NoArgsConstructor @ToString @Slf4j
061public class UpdateProjectLicenseMojo extends AbstractLicenseMojo {
062
063    /**
064     * {@link #SPDX_LICENSE_IDENTIFIER} = {@value #SPDX_LICENSE_IDENTIFIER}
065     */
066    protected static final String SPDX_LICENSE_IDENTIFIER = "SPDX-License-Identifier";
067
068    @Parameter(defaultValue = "${project.licenses[0].name}", property = "license.name")
069    private String name = null;
070
071    @Parameter(defaultValue = "${project.licenses[0].url}", property = "license.url")
072    private String url = null;
073
074    @Parameter(defaultValue = "false", property = "license.download")
075    private boolean download = false;
076
077    @Parameter(defaultValue = "true", property = "license.overwrite")
078    private boolean overwrite = true;
079
080    @Parameter(defaultValue = "true", property = "license.verify")
081    private boolean verify = true;
082
083    @Inject private MavenProject project = null;
084    @Inject private ArtifactLicenseCatalog catalog = null;
085    @Inject private LicenseResolver resolver = null;
086
087    @Override
088    public void execute() throws MojoExecutionException, MojoFailureException {
089        if (name == null && url == null) {
090            throw new MojoFailureException("One of license 'name' or 'url' must be specified");
091        }
092
093        if (! isSkip()) {
094            /*
095             * Find the license from name and/or URL, ...
096             */
097            AnyLicenseInfo license = null;
098
099            if (license == null) {
100                List<AnyLicenseInfo> list =
101                    Stream.of(project.getLicenses())
102                    .filter(Objects::nonNull)
103                    .flatMap(List::stream)
104                    .map(t -> new URLLicenseInfo(t.getName(), t.getUrl()))
105                    .map(t -> resolver.parse(t))
106                    .collect(toList());
107
108                license = resolver.toLicense(list);
109
110                if (! LicenseUtilityMethods.isFullySpdxListed(license)) {
111                    warnIfExtractedLicenseInfo(Stream.of(license));
112                }
113            }
114
115            if (license != null) {
116                if (LicenseUtilityMethods.isFullySpdxListed(license)) {
117                    String id = license.toString();
118
119                    if (id != null && (! id.equals(name))) {
120                        log.warn("{}: {}", SPDX_LICENSE_IDENTIFIER, id);
121                    }
122                } else {
123                }
124            }
125            /*
126             * ... update the project copy, ...
127             */
128            if (download || (! getFile().exists())) {
129                if (url != null) {
130                    copy(url, getFile().toPath(), overwrite);
131                }
132            }
133            /*
134             * ... verify any local copy is actually the specified license (and
135             * fail if it is not), ...
136             */
137            if (verify) {
138                if (LicenseUtilityMethods.isFullySpdxListed(license)) {
139                    /*
140                     * TBD: Bad assumption.
141                     */
142                    if (license instanceof License) {
143                        try {
144                            Document document = Jsoup.parse(getFile(), null);
145
146                            document.outputSettings()
147                                .syntax(Document.OutputSettings.Syntax.xml);
148
149                            boolean isDifferenceFound =
150                                isTextStandardLicense((License) license, document.body().text())
151                                .isDifferenceFound();
152
153                            if (isDifferenceFound) {
154                                fail(getFile() + " does not contain "
155                                     + ((License) license).getLicenseId() + " license text");
156                            }
157                        } catch (MojoFailureException exception) {
158                            throw exception;
159                        } catch (Exception exception) {
160                            fail("Cannot analyze " + getFile(), exception);
161                        }
162                    }
163                }
164            }
165            /*
166             * ... update the project POM if necessary, ...
167             */
168            if (license instanceof License) {
169                File pom = project.getFile();
170                Model model = null;
171
172                try (FileInputStream in = new FileInputStream(pom)) {
173                    model = new MavenXpp3Reader().read(in);
174                } catch (Exception exception) {
175                    fail("Cannot read " + pom, exception);
176                }
177
178                if (isBlank(model.getInceptionYear())
179                    || (model.getLicenses() == null || model.getLicenses().size() != 1)
180                    || (! ((License) license).getLicenseId().equals(name))) {
181                    /*
182                     * org.apache.maven.model.License update =
183                     *     project.getLicenses().get(0).clone();
184                     *
185                     * update.setName(license.getLicenseId());
186                     *
187                     * model.setLicenses(Arrays.asList(update));
188                     *
189                     * File backup =
190                     *     new File(pom.getParentFile(), pom.getName() + ".bak");
191                     *
192                     * pom.renameTo(backup);
193                     *
194                     * try (FileOutputStream out = new FileOutputStream(pom)) {
195                     *     new MavenXpp3Writer().write(out, model);
196                     * } catch (Exception exception) {
197                     *     fail("Cannot write " + pom, exception);
198                     * }
199                     */
200                }
201            }
202            /*
203             * ... and record the result.
204             */
205            if (license != null) {
206                catalog.put(project.getArtifact(), license);
207                catalog.flush();
208            }
209        } else {
210            log.info("Skipping {} update", getFile().getName());
211        }
212    }
213}

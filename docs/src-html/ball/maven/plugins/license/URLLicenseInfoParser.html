072/**
073 * {@link URL} ({@link String} representation) to {@link AnyLicenseInfo}
074 * {@link java.util.Map} implementation.  The {@link #get(Object)} method
075 * transparently calculates and caches any value.
076 *
077 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
078 */
079@Named @Singleton
080@Slf4j
081public class URLLicenseInfoParser extends ConcurrentSkipListMap<String,AnyLicenseInfo> implements DefaultMethods {
082    private static final long serialVersionUID = -1086861055455360074L;

084    private static final HostnameVerifier NONE = new HostnameVerifierImpl();
085
086    private static final Set<Integer> REDIRECT_CODES =
087        Stream.of(HttpURLConnection.HTTP_MOVED_TEMP,
088                  HttpURLConnection.HTTP_MOVED_PERM,
089                  HttpURLConnection.HTTP_SEE_OTHER)
090        .collect(toSet());
091
092    private static final Pattern CANONICAL = Pattern.compile("<([^>]*)>; rel=\"canonical\"");
093
094    /** @serial */ private final LicenseMap map;
095    /** @serial */ private final Map<Pattern,String> redirects;

097    /**
098     * Sole constructor.
099     *
100     * @param   map             The injected {@link LicenseMap}.
101     */
102    @Inject
103    public URLLicenseInfoParser(LicenseMap map) {
104        super(String.CASE_INSENSITIVE_ORDER);
105
106        this.map = Objects.requireNonNull(map);
107
108        try {
109            for (AnyLicenseInfo value : map.values()) {
110                if (value instanceof License) {
111                    String id = ((License) value).getLicenseId();
112
113                    put(String.format("https://opensource.org/licenses/%s", id), value);
114                    put(String.format("https://spdx.org/licenses/%s.html", id), value);
115                }
116            }
117
118            for (JsonNode node : LicenseMap.LICENSES_FULL_JSON.at("/licenses")) {
119                AnyLicenseInfo value = map.get(node.at("/identifiers/spdx/0").asText());
120
121                if (value != null) {
122                    for (JsonNode uri : node.at("/uris")) {
123                        String key = uri.asText();
124
125                        putIfAbsent(key, value);
126                    }
127                }
128            }
129
130            Properties seeds = getXMLProperties("seeds");
131
132            for (String id : seeds.stringPropertyNames()) {
133                AnyLicenseInfo value = LicenseInfoFactory.parseSPDXLicenseString(id);
134
135                if (! isFullySpdxListed(value)) {
136                    throw new IllegalArgumentException(id);
137                }
138
139                Stream.of(seeds.getProperty(id).split("[\\p{Space}]+"))
140                    .filter(StringUtils::isNotBlank)
141                    .forEach(t -> putIfAbsent(t.trim(), value));
142            }
143
144            for (AnyLicenseInfo value : map.values()) {
145                if (value instanceof License) {
146                    for (String key : ((License) value).getSeeAlso()) {
147                        if (isNotBlank(key)) {
148                            putIfAbsent(key.trim(), value);
149                        }
150                    }
151                }
152            }
153
154            Set<String> set =
155                keySet().stream()
156                .filter(t -> t.startsWith("https:"))
157                .collect(toSet());
158
159            set.stream()
160                .forEach(t -> computeIfAbsent(t.replace("https:", "http:"), k -> get(t)));
161
162            redirects =
163                getXMLProperties("redirects").entrySet().stream()
164                .collect(toMap(k -> Pattern.compile(k.getKey().toString()),
165                               v -> v.getValue().toString().trim()));
166        } catch (Exception exception) {
167            log.error("{}", exception.getMessage(), exception);
168            throw new ExceptionInInitializerError(exception);
169        }
170    }

180    /**
181     * Entry-point that first attempts to parse the specified license ID as
182     * an SPDX license ID and, if that fails, continues to analyze the
183     * document at specified URL.
184     *
185     * @param   resolver        The {@link LicenseResolver}.
186     * @param   key             The {@link URLLicenseInfo}.
187     *
188     * @return  {@link AnyLicenseInfo} reppresenting the results of the
189     *          parse (may be {@code key}).
190     */
191    public AnyLicenseInfo parse(LicenseResolver resolver, URLLicenseInfo key) {
192        String name = key.getLicenseId();
193        AnyLicenseInfo value = isNotBlank(name) ? map.get(name) : null;
194
195        if (value == null) {
196            Set<AnyLicenseInfo> set =
197                Stream.of(key.getSeeAlso())
198                .map(t -> computeIfAbsent(t, k -> compute(resolver, k)))
199                .collect(toSet());
200
201            if (! set.isEmpty()) {
202                value = resolver.toLicense(set);
203            }
204        }
205
206        return (value != null) ? value : key;
207    }

209    private AnyLicenseInfo compute(LicenseResolver resolver, String url) {
210        AnyLicenseInfo value = null;
211        URLConnection connection = null;
212
213        try {
214            connection = new URL(url).openConnection();
215
216            if (connection instanceof HttpURLConnection) {
217                ((HttpURLConnection) connection).setInstanceFollowRedirects(false);
218            }
219
220            if (connection instanceof HttpsURLConnection) {
221                ((HttpsURLConnection) connection).setHostnameVerifier(NONE);
222            }
223
224            String canonicalURL = getCanonicalURL(connection);
225
226            if (isNotBlank(canonicalURL) && (! equals(canonicalURL, url))) {
227                if (value == null) {
228                    value = get(canonicalURL);
229                }
230            }
231
232            String redirectURL = getRedirectURL(connection);
233
234            if (isNotBlank(redirectURL)) {
235                if (! equals(redirectURL, url)) {
236                    if (value != null) {
237                        put(redirectURL, value);
238                    } else {
239                        value = computeIfAbsent(redirectURL, k -> compute(resolver, k));
240                    }
241                }
242            }
243
244            if (value == null) {
245                ContentType type = ContentType.parse(connection.getContentType());
246                Charset charset = (type.getCharset() != null) ? type.getCharset() : UTF_8;
247
248                if (type.getMimeType().matches("(?i).*(html|xml).*")) {
249                    try (InputStream in = connection.getInputStream()) {
250                        Document document = Jsoup.parse(in, charset.name(), url);
251
252                        document.outputSettings()
253                            .syntax(Document.OutputSettings.Syntax.xml);
254                        /*
255                         * Heuristic: Look for a "canonical" <link/> with a
256                         * known href.
257                         */
258                        if (value == null) {
259                            value =
260                                document.select("head>link[rel='canonical'][href]").stream()
261                                .map(t -> t.attr("abs:href"))
262                                .filter(StringUtils::isNotBlank)
263                                .map(t -> get(t))
264                                .filter(Objects::nonNull)
265                                .findFirst().orElse(null);
266                        }
267                        /*
268                         * Heuristics to consider:
269                         *
270                         * Search for SPDX-License-Identifier
271                         *
272                         * If a single href of
273                         * http://opensource.org/licenses/([^/]) is found
274                         * and $1 is in the LicenseMap
275                         */
276                        /*
277                         * Parse the document if the heuristics fail.
278                         */
279                        if (value == null) {
280                            value =
281                                Stream.of("content, .content, #content", "main, .main, #main", "body, .body, #body")
282                                .map(t -> document.select(t))
283                                .flatMap(Elements::stream)
284                                .filter(Element::hasText)
285                                .map(t -> t.text())
286                                .distinct()
287                                .map(t -> new TextLicenseInfo(url, t, url))
288                                .map(t -> resolver.parse(t))
289                                .filter(Objects::nonNull)
290                                .filter(t -> (! (t instanceof ExtractedLicenseInfo)))
291                                .findFirst()
292                                .orElse(null);
293                        }
294
295                        if (value == null) {
296                            value = new TextLicenseInfo(url, document.wholeText(), url);
297                            value = resolver.parse(value);
298                        }
299                    }
300                } else {
301                    try (InputStream in = connection.getInputStream();
302                         Scanner scanner = new Scanner(in, charset.name())) {
303                        String text = scanner.useDelimiter("\\A").next();
304
305                        value = new TextLicenseInfo(url, text, url);
306                        value = resolver.parse(value);
307                    }
308                }
309            }
310        } catch (FileNotFoundException exception) {
311            log.debug("File not found: {}", url);
312        } catch (Exception exception) {
313            log.warn("Cannot read {}", url);
314
315            if (connection != null) {
316                connection.getHeaderFields().entrySet().stream()
317                    .forEach(t -> log.debug("{}", t));
318            }
319        } finally {
320            if (value == null) {
321                value = new TextLicenseInfo(url, EMPTY, url);
322            }
323        }
324
325        if (value instanceof ExtractedLicenseInfo) {
326            TextLicenseInfo.addSeeAlso((ExtractedLicenseInfo) value, url);
327        }
328
329        return value;
330    }

332    private boolean equals(String left, String right) {
333        return (Objects.compare(left, right, comparator()) == 0 && Objects.compare(right, left, comparator()) == 0);
334    }

336    private String getCanonicalURL(URLConnection connection) {
337        String url = null;
338
339        if (connection instanceof HttpURLConnection) {
340            List<String> list = connection.getHeaderFields().get("Link");
341
342            if (list != null) {
343                url =
344                    list.stream()
345                    .map(t -> CANONICAL.matcher(t))
346                    .filter(t -> t.find())
347                    .map(t -> t.group(1))
348                    .filter(StringUtils::isNotBlank)
349                    .map(t -> resolve((HttpURLConnection) connection, t))
350                    .map(t -> t.toASCIIString())
351                    .findFirst().orElse(null);
352            }
353        }
354
355        return url;
356    }
357
358    private String getRedirectURL(URLConnection connection) throws IOException {
359        String redirectURL = null;
360
361        if (connection instanceof HttpURLConnection) {
362            String url = connection.getURL().toString();
363            int code = ((HttpURLConnection) connection).getResponseCode();
364
365            if (REDIRECT_CODES.contains(code)) {
366                String location = connection.getHeaderField("Location");
367
368                if (isNotBlank(location)) {
369                    redirectURL = resolve((HttpURLConnection) connection, location).toASCIIString();
370                }
371            }
372
373            if (isBlank(redirectURL)) {
374                for (Map.Entry<Pattern,String> entry : redirects.entrySet()) {
375                    Matcher matcher = entry.getKey().matcher(url);
376
377                    if (matcher.matches()) {
378                        redirectURL = matcher.replaceFirst(entry.getValue());
379
380                        if (isNotBlank(redirectURL)) {
381                            break;
382                        } else {
383                            redirectURL = null;
384                        }
385                    }
386                }
387            }
388        }
389
390        return redirectURL;
391    }
392
393    private URI resolve(HttpURLConnection connection, String location) {
394        URI uri = URI.create(location);
395
396        if (! uri.isAbsolute()) {
397            try {
398                uri = connection.getURL().toURI().resolve(uri);
399            } catch (Exception exception) {
400                log.debug("{}", exception.getMessage(), exception);
401            }
402        }
403
404        return uri;
405    }

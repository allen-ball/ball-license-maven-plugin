049/**
050 * Abstract base class for license {@link org.apache.maven.plugin.Mojo}s.
051 *
052 * {@injected.fields}
053 *
054 * {@bean.info}
055 *
056 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
057 */
058@NoArgsConstructor(access = PROTECTED) @Getter @ToString @Slf4j
059public abstract class AbstractLicenseMojo extends AbstractMojo {
060    @Parameter(defaultValue = "false", property = "license.skip")
061    private boolean skip = false;
062
063    @Parameter(defaultValue = "${basedir}/LICENSE", property = "license.file", readonly = true)
064    private File file = null;
065
066    protected void warnIfExtractedLicenseInfo(Stream<AnyLicenseInfo> stream) {
067        Set<ExtractedLicenseInfo> extracted =
068            stream
069            .flatMap(t -> LicenseUtilityMethods.walk(t))
070            .filter(t -> (t instanceof ExtractedLicenseInfo))
071            .map(t -> (ExtractedLicenseInfo) t)
072            .collect(toSet());
073
074        if (! extracted.isEmpty()) {
075            log.warn("Cannot match to SPDX license(s)");
076
077            for (ExtractedLicenseInfo license : extracted) {
078                String id = license.getLicenseId();
079
080                log.warn("    '{}'", license.getLicenseId());
081
082                String[] seeAlso = license.getSeeAlso();
083
084                if (seeAlso != null) {
085                    for (String string : seeAlso) {
086                        if (! string.equals(license.getLicenseId())) {
087                            log.warn("        {}", string);
088                        }
089                    }
090                }
091            }
092        }
093    }
094
095    /**
096     * Method to copy from {@link URL} ({@link String} representation) to
097     * local {@link Path}.
098     *
099     * @param   from            The source {@link URL} ({@link String}
100     *                          representation).
101     * @param   to              The target {@link Path}.
102     * @param   overwrite       Whether or not to overwrite an existing
103     *                          {@link Path}.
104     *
105     * @throws  MojoExecutionException
106     *                          See
107     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
108     * @throws  MojoFailureException
109     *                          See
110     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
111     */
112    protected void copy(String from, Path to, boolean overwrite) throws MojoExecutionException, MojoFailureException {
113        try {
114            copy(new URL(from), to, overwrite);
115        } catch (MalformedURLException exception) {
116            fail(from, exception);
117        }
118    }
119
120    /**
121     * Method to copy from {@link URL} ({@link String} representation) to
122     * local {@link Path}.
123     *
124     * @param   from            The source {@link URL}.
125     * @param   to              The target {@link Path}.
126     * @param   overwrite       Whether or not to overwrite an existing
127     *                          {@link Path}.
128     *
129     * @throws  MojoExecutionException
130     *                          See
131     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
132     * @throws  MojoFailureException
133     *                          See
134     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
135     */
136    protected void copy(URL from, Path to, boolean overwrite) throws MojoExecutionException, MojoFailureException {
137        String message = null;
138
139        try {
140            message = to.toString() + " <- " + from.toString();
141
142            if (Files.isDirectory(to)) {
143                throw new FileAlreadyExistsException(to.toString(), null, "Is not a file or link");
144            }
145
146            FileTime local = FileTime.fromMillis(0);
147
148            if (Files.exists(to)) {
149                local = Files.getLastModifiedTime(to);
150            }
151
152            URLConnection connection = from.openConnection();
153            FileTime remote = FileTime.fromMillis(connection.getLastModified());
154            boolean isNewer = (! Files.exists(to)) || remote.toMillis() == 0 || remote.compareTo(local) > 0;
155
156            if (isNewer) {
157                CopyOption[] options =
158                    Stream.of(StandardCopyOption.REPLACE_EXISTING)
159                    .filter(t -> isNewer & overwrite)
160                    .toArray(CopyOption[]::new);
161
162                try (InputStream in = connection.getInputStream()) {
163                    Files.copy(in, to, options);
164                }
165
166                if (remote.toMillis() != 0) {
167                    Files.setLastModifiedTime(to, remote);
168                }
169
170                log.info("{}", message);
171            } else {
172                log.info("{} is up-to-date", to);
173            }
174        } catch (IOException exception) {
175            fail(message, exception);
176        } catch (Throwable throwable) {
177            log.error("{}", throwable.getMessage(), throwable);
178
179            if (throwable instanceof MojoExecutionException) {
180                throw (MojoExecutionException) throwable;
181            } else if (throwable instanceof MojoFailureException) {
182                throw (MojoFailureException) throwable;
183            } else {
184                throw new MojoExecutionException(throwable.getMessage(),
185                                                 throwable);
186            }
187        }
188    }
189
190    /**
191     * Log and throw a {@link MojoFailureException}.
192     *
193     * @param   message         The message {@link String}.
194     *
195     * @throws  MojoFailureException
196     *                          Always.
197     */
198    protected void fail(String message) throws MojoFailureException {
199        log.error("{}", message);
200        throw new MojoFailureException(message);
201    }
202
203    /**
204     * Log and throw a {@link MojoFailureException}.
205     *
206     * @param   message         The message {@link String}.
207     * @param   reason          The reason {@link Throwable}.
208     *
209     * @throws  MojoFailureException
210     *                          Always.
211     */
212    protected void fail(String message, Throwable reason) throws MojoFailureException {
213        log.error("{}", message, reason);
214        throw new MojoFailureException(message, reason);
215    }
216}
